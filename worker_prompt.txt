import requests

BASE_URL = "https://api.example.com"

def get_user(user_id):
    response = requests.get(f"{BASE_URL}/users/{user_id}")
    response.raise_for_status()
    return response.json()

def create_user(data):
    response = requests.post(f"{BASE_URL}/users", json=data)
    response.raise_for_status()
    return response.json()

def update_user(user_id, data):
    response = requests.put(f"{BASE_URL}/users/{user_id}", json=data)
    response.raise_for_status()
    return response.json()

def delete_user(user_id):
    response = requests.delete(f"{BASE_URL}/users/{user_id}")
    response.raise_for_status()
    return response.status_code

## What to change
- Replace `import requests` with `import httpx`
- Convert every function to `async def`
- Use `async with httpx.AsyncClient() as client:` pattern for each function
- Map: requests.get → client.get, requests.post → client.post, requests.put → client.put, requests.delete → client.delete

## Expected result
import httpx

BASE_URL = "https://api.example.com"

async def get_user(user_id):
    async with httpx.AsyncClient() as client:
        response = await client.get(f"{BASE_URL}/users/{user_id}")
        response.raise_for_status()
        return response.json()

async def create_user(data):
    async with httpx.AsyncClient() as client:
        response = await client.post(f"{BASE_URL}/users", json=data)
        response.raise_for_status()
        return response.json()

async def update_user(user_id, data):
    async with httpx.AsyncClient() as client:
        response = await client.put(f"{BASE_URL}/users/{user_id}", json=data)
        response.raise_for_status()
        return response.json()

async def delete_user(user_id):
    async with httpx.AsyncClient() as client:
        response = await client.delete(f"{BASE_URL}/users/{user_id}")
        response.raise_for_status()
        return response.status_code
